ประวัติเอกสาร

เวลาเหตุการณ์ไฟล์ผู้ใช้งานรายละเอียด

ตั้งค่าการเชื่อมต่อใช้งานจริง

กรอกข้อมูลจาก Google Cloud และ Supabase แล้วกดบันทึก จากนั้นกดเชื่อมต่อ Google ใช้ได้กับ GitHub Pages, Cloudflare Pages, Netlify หรือ Vercel

หมายเหตุ: Client-side app ใช้ Access Token แบบอายุสั้น หาก Token หมดอายุให้กดเชื่อมต่อ Google อีกครั้ง

ขั้นตอนก่อนนำขึ้นเว็บ

  1. เปลี่ยนชื่อไฟล์นี้เป็น index.html
  2. นำไฟล์ขึ้น Hosting เช่น Cloudflare Pages / GitHub Pages / Netlify / Vercel
  3. นำ URL เว็บจริงไปใส่ใน Google Cloud: Authorized JavaScript origins
  4. สร้างโฟลเดอร์ใน Google Drive แล้วคัดลอก Folder ID มาใส่ในหน้านี้
  5. สร้างตาราง Supabase ด้วย SQL ด้านล่าง แล้วนำ Supabase URL และ anon key มาใส่
  6. กดบันทึกการตั้งค่า → เชื่อมต่อ Google → ทดสอบ Supabase → Upload PDF

หมายเหตุ: ห้ามใส่ Google Client Secret หรือ Supabase service_role key ในไฟล์หน้าเว็บเด็ดขาด ให้ใช้เฉพาะ OAuth Client ID และ anon/public key เท่านั้น

SQL สำหรับสร้างตาราง Supabase

create extension if not exists pgcrypto;

create table if not exists public.documents (
  id uuid primary key default gen_random_uuid(),
  farmer_code text,
  farmer_name text,
  crop_year text,
  document_type text,
  file_name text not null,
  drive_file_id text,
  drive_link text,
  status text default 'Pending',
  note text,
  uploaded_by text,
  signed_by text,
  created_at timestamptz default now(),
  signed_at timestamptz
);

create table if not exists public.document_history (
  id uuid primary key default gen_random_uuid(),
  document_id uuid references public.documents(id) on delete cascade,
  event_type text not null,
  file_name text,
  actor text,
  detail text,
  created_at timestamptz default now()
);

create index if not exists idx_documents_farmer_code on public.documents(farmer_code);
create index if not exists idx_documents_farmer_name on public.documents(farmer_name);
create index if not exists idx_documents_status on public.documents(status);
